## âš™ï¸ Example Code â€” `FloatingShapes.jsx`

```jsx
"use client";
import { motion } from "framer-motion";
import { useEffect, useState } from "react";

export default function FloatingShapes() {
  const [mouse, setMouse] = useState({ x: 0, y: 0 });

  useEffect(() => {
    const handleMouseMove = (e) => {
      setMouse({ x: e.clientX, y: e.clientY });
    };
    window.addEventListener("mousemove", handleMouseMove);
    return () => window.removeEventListener("mousemove", handleMouseMove);
  }, []);

  return (
    <div className="fixed right-0 top-0 h-full w-1/3 flex flex-col items-center justify-center gap-10 pointer-events-none">
      {/* Circle */}
      <motion.div
        className="w-24 h-24 rounded-full bg-blue-400/40 blur-md"
        animate={{
          x: (mouse.x - window.innerWidth / 2) * 0.05,
          y: (mouse.y - window.innerHeight / 2) * 0.05,
          rotate: mouse.x * 0.05,
        }}
        transition={{ type: "spring", stiffness: 50, damping: 20 }}
      />

      {/* Square */}
      <motion.div
        className="w-20 h-20 bg-pink-400/40 rotate-45 blur-sm"
        animate={{
          x: (mouse.x - window.innerWidth / 2) * -0.04,
          y: (mouse.y - window.innerHeight / 2) * 0.04,
          rotate: mouse.y * -0.03,
        }}
        transition={{ type: "spring", stiffness: 50, damping: 25 }}
      />

      {/* Triangle */}
      <motion.div
        className="w-0 h-0 border-l-[40px] border-r-[40px] border-b-[70px] border-l-transparent border-r-transparent border-b-yellow-300/50 blur-sm"
        animate={{
          x: (mouse.x - window.innerWidth / 2) * 0.03,
          y: (mouse.y - window.innerHeight / 2) * -0.03,
          rotate: mouse.x * 0.02,
        }}
        transition={{ type: "spring", stiffness: 60, damping: 25 }}
      />
    </div>
  );
}
```
